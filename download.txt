It uses hmac.compare_digest to prevent timing attacks at the Auth Gate, and utilizes subprocess.Popen to instantly detach the NanoClaw container process, ensuring GitHub receives a 202 Accepted response within milliseconds.

import os
import hmac
import hashlib
import logging
import subprocess
from fastapi import FastAPI, Request, HTTPException, status
from dotenv import load_dotenv

# State 0: IDLE_LISTEN Environment Setup
load_dotenv()
GITHUB_WEBHOOK_SECRET = os.getenv("GITHUB_WEBHOOK_SECRET")

if not GITHUB_WEBHOOK_SECRET:
   raise ValueError("CRITICAL: GITHUB_WEBHOOK_SECRET not found in .env")

app = FastAPI(title="EN-OS Event Bridge")
logging.basicConfig(level=logging.INFO, format="%(asctime)s | %(levelname)s | %(message)s")
logger = logging.getLogger("EN-OS.Daemon")

def verify_signature(payload_body: bytes, signature_header: str) -> bool:
   """State 1: AUTH_GATE - Cryptographic validation of GitHub payload."""
   if not signature_header:
       return False
   hash_object = hmac.new(
       GITHUB_WEBHOOK_SECRET.encode("utf-8"), 
       msg=payload_body, 
       digestmod=hashlib.sha256
   )
   expected_signature = "sha256=" + hash_object.hexdigest()
   # compare_digest prevents timing analysis attacks
   return hmac.compare_digest(expected_signature, signature_header)

def ignite_nanoclaw(issue_number: int, repo_name: str):
   """State 3: NANOCLAW_IGNITION - Detached compute cycle."""
   logger.info(f"IGNITION: Booting NanoClaw for {repo_name}#{issue_number}")
   
   cmd = [
       "docker", "run", "--rm",
       "--env-file", ".env",
       "-e", f"TARGET_ISSUE={issue_number}",
       "-e", f"TARGET_REPO={repo_name}",
       "nanoclaw:latest"
   ]
   
   try:
       # Popen detaches the process. The daemon drops the reference and returns to State 0.
       # Output is routed to DEVNULL to prevent local buffer lockups. 
       # The container must log its own forensic telemetry to ChromaDB/Registry.
       subprocess.Popen(cmd, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)
   except Exception as e:
       logger.error(f"IGNITION FAILED: {str(e)}")

@app.post("/webhook", status_code=status.HTTP_202_ACCEPTED)
async def github_webhook(request: Request):
   payload_body = await request.body()
   signature = request.headers.get("x-hub-signature-256")
   event_type = request.headers.get("x-github-event")

   # --- STATE 1: AUTH GATE ---
   if not verify_signature(payload_body, signature):
       logger.warning("AUTH GATE FAILED: Invalid or missing signature.")
       raise HTTPException(
           status_code=status.HTTP_401_UNAUTHORIZED, 
           detail="Signature validation failed"
       )

   # --- STATE 2: PAYLOAD PARSE ---
   payload = await request.json()
   target_issue_number = None
   target_repo = payload.get("repository", {}).get("full_name", "mechanistic-org/global_agent")

   # Trigger A: Explicit manual override via comment
   if event_type == "issue_comment" and payload.get("action") == "created":
       comment_body = payload.get("comment", {}).get("body", "")
       if "/execute" in comment_body.lower():
           target_issue_number = payload.get("issue", {}).get("number")
           logger.info(f"TRIGGER: Manual /execute detected on #{target_issue_number}")

   # Trigger B: Automated board movement (Sprint Now)
   elif event_type == "projects_v2_item" and payload.get("action") == "edited":
       # Note: GitHub Projects V2 payloads are heavily nested. 
       # You must inspect the specific JSON shape of your 'Sprint Now' column move.
       changes = payload.get("changes", {})
       if "field_value" in changes:
           field_name = changes["field_value"].get("field_name")
           to_value = changes["field_value"].get("to", {}).get("name")
           
           if field_name == "Status" and to_value == "Sprint Now":
               # Extract issue number from the content_node or project item URL
               # This logic may require adjustment based on the exact GraphQL node map
               content_url = payload.get("projects_v2_item", {}).get("content_node_id")
               logger.info(f"TRIGGER: Item moved to Sprint Now. Node: {content_url}")
               
               # Placeholder: If the flat issue number isn't in the payload, 
               # you will need a quick gh api call here to resolve the node_id to an issue integer.
               # target_issue_number = resolve_node_to_issue(content_url) 
               
   # --- STATE 3: NANOCLAW IGNITION ---
   if target_issue_number:
       ignite_nanoclaw(target_issue_number, target_repo)
       return {"status": "accepted", "message": f"NanoClaw ignition queued for #{target_issue_number}"}

   # State transition failed (Noise). Acknowledge and drop.
   return {"status": "ignored", "message": "Payload did not match ignition triggers."}

Engineering Notes on Execution
1. Process Detachment (subprocess.Popen): By using Popen without .wait() or communicate(), the Python daemon fires the command to the Docker daemon and instantly drops it. This guarantees the webhook returns a 202 before GitHub's strict 10-second timeout window closes, preventing red "timeout" errors in your repository settings.
2. State 4 Enforcement: Note that stdout and stderr are dumped to DEVNULL. The daemon does not babysit the container. If the agent fails, it must use its internal /session_close tools to write that failure to GitHub or ChromaDB. This enforces the isolation of the Ephemeral Node.
3. Project V2 Nesting: Standard Issue webhooks are flat and easy to parse. Project V2 webhooks (Trigger B) are GraphQL-backed and notoriously obtuse. You will likely need to inspect a raw payload from Project #5 to write the exact JSON traversal path to extract the issue number.
