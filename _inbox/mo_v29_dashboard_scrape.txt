            </p>
          </div>

          <div>
            <h4 className="mb-2 text-heading-sm text-white">
              D. Auto-Acceptance Protocol
            </h4>
            <div className="border-l-2 border-blue-500 bg-[#111820] p-4 font-mono text-body-primary">

                {data.engagementTerms.acceptanceCriteria?.clause || "Deliverables are deemed accepted if no written rejection is received within 5 business days of submission. Payment is not contingent on hardware yield."}

            </div>
          </div>

          <div>
            <h4 className="mb-2 text-heading-sm text-white">
              E. Payment Terms
            </h4>
            <p className="border-l-2 border-emerald-500 bg-[#111820] p-4 font-mono text-body-primary">
              {data.engagementTerms.paymentTerms}
            </p>
          </div>

          <div>
            <h4 className="mb-2 text-heading-sm text-white">
              F. Termination & Execution
            </h4>
            <p className="border-l-2 border-neutral-500 bg-[#111820] p-4 font-mono text-body-primary">
              {data.engagementTerms.termination}
            </p>
          </div>
        </div>
      </div>

      {/* Compensation Structure */}
      {vaultF && (
        <div className="mt-8 border border-neutral-800 bg-[#0d1117] p-6 lg:p-10">
          <div className="mb-8 border-b border-neutral-800 pb-4">
            <h3 className="flex items-center gap-3 text-heading-base text-emerald-500">
              <span className="flex h-8 w-8 items-center justify-center bg-emerald-500/10 text-sm">
                III
              </span>
              COMPENSATION STRUCTURE
            </h3>
          </div>

          <div className="flex flex-col gap-8">
            {/* A. Evergreen Retainer - Full Width */}
            <div className="w-full">
              <h4 className="mb-2 text-heading-sm text-white">
                A. Evergreen Retainer
              </h4>
              <div className="flex flex-col gap-4 border-l-2 border-emerald-500 bg-[#111820] p-4 text-base">
                <div>
                  <div className="font-mono text-2xl font-bold text-emerald-500 mb-2">
                    {vaultF.roiJustification.quarterlyRetainerLabel}
                  </div>
                </div>

                {vaultF.roiJustification.evergreenTerms && (
                  <div className="mt-1 border-t border-neutral-800/50 pt-3">
                    <div className="mb-1 text-label text-white flex items-center gap-2">
                       <span className="w-2 h-2 rounded-full bg-emerald-500 block"></span>

                    <p className="text-body-secondary mb-2">
                      {vaultF.roiJustification.evergreenTerms.overview}

            <div className="grid gap-8 lg:grid-cols-2">
              {vaultF.advisoryEquityVesting && (
                <div className="w-full">
                  <h4 className="mb-2 text-heading-sm text-white">
                    B. Equity
                  </h4>
                  <div className="flex flex-col gap-4 border-l-2 border-amber-500 bg-[#111820] p-4 text-base h-full">
                    <div>
                      <div className="font-mono text-xl xl:text-2xl font-bold text-amber-500">
                        Frontier Science - Data Moat - Advisory Equity: 0.5%
                      </div>
                      <div className="mt-2 text-body-secondary">
                        {vaultF.advisoryEquityVesting.description}
                      </div>
                    </div>

                    <div className="flex flex-col gap-2 border-t border-neutral-800 pt-4 mt-2">
                      {vaultF.advisoryEquityVesting.milestoneTranches.map((t) => (
                        <div key={t.tranche} className="text-base text-neutral-400 flex flex-col sm:flex-row gap-1 sm:gap-2">
                          <span className="font-bold text-amber-500 whitespace-nowrap">Tranche {t.tranche} ({t.percentageUnlock}%):</span>
                          <span className="leading-relaxed">{t.milestone}</span>
                        </div>
                      ))}
                    </div>
                  </div>
                </div>
              )}

              {vaultF.executiveLeverage && vaultF.executiveLeverage.gates && (
                <div className="w-full">
                  <h4 className="mb-2 text-heading-sm text-white">
                    C. Milestone Bonuses
                  </h4>
                  <div className="flex flex-col gap-4 border-l-2 border-blue-500 bg-[#111820] p-4 text-base h-full">

                      <div className="font-mono text-2xl font-bold text-blue-500">

                      <div className="mt-2 text-body-secondary text-sm">

                          ? new Intl.NumberFormat("en-US", {

                                    ? "border-amber-500/30 bg-amber-500/10 font-bold text-amber-500"
                                    : "border-neutral-500/30 bg-neutral-500/10 font-mono italic text-neutral-400"
                                }`}
                              >
                                {formattedBonus}
                              </div>
                            </summary>
                            <div className="mt-3 border-l-2 border-blue-500/30 pl-4 text-sm leading-relaxed text-neutral-400">
                              {g.description}
                            </div>
                          </details>
                        );
                      })}
                    </div>
                  </div>
                </div>
              )}
            </div>
          </div>

          {/* UNIFIED COMPENSATION DASHBOARD */}
          <div className="mt-12 border-t border-neutral-800 pt-8">
            <h4 className="mb-6 text-heading-sm text-white tracking-widest flex items-center gap-2">
              <span className="h-2 w-2 bg-emerald-500 block"></span>
              COMPENSATION STRUCTURE DASHBOARD
            </h4>

            <div className="mb-8">
              <RoiSankeyDiagram vaultF={vaultF as any} />
            </div>
            
            {/* 1. Evergreen Retainer */}
            <div className="mb-8">
              <div className="text-[10px] text-neutral-500 mb-4 tracking-widest border-b border-neutral-800 pb-2">
                EVERGREEN RETAINER (ACTIVE FUNDING)
              </div>
              <div className="flex flex-col sm:flex-row sm:items-center justify-between p-3 border border-neutral-800 bg-[#0d1117]">
                <div className="flex items-center gap-3 mb-2 sm:mb-0">
                  <div className="w-4 h-4 border border-neutral-600 shrink-0"></div>
                  <div>
                    <span className="text-[10px] text-neutral-500 mr-2">RETAINER_BASE</span>
                    <span className="font-bold text-neutral-300">Quarterly Intervention Protocol</span>
                  </div>
                </div>
                <div className="flex items-center gap-4 text-[10px] pl-7 sm:pl-0">
                  <span className="text-emerald-500 font-bold text-xs">$87,500 / QTR</span>
                  <span className="px-2 py-0.5 font-bold tracking-widest bg-neutral-800 text-neutral-500">PENDING</span>
                </div>
              </div>
            </div>

            {/* 2. Advisory Equity */}
            {vaultF.advisoryEquityVesting && (
              <div className="mb-8">
                <div className="text-[10px] text-neutral-500 mb-4 tracking-widest border-b border-neutral-800 pb-2">
                  ADVISORY EQUITY VESTING TRANCHES (API TRIGGERS)
                </div>
                <div className="space-y-2">
                  {vaultF.advisoryEquityVesting.milestoneTranches.map((t, idx) => (
                    <div key={idx} className="flex flex-col sm:flex-row sm:items-center justify-between p-3 border border-neutral-800 bg-[#0d1117]">
                      <div className="flex items-center gap-3 mb-2 sm:mb-0">
                        <div className="w-4 h-4 border border-neutral-600 shrink-0"></div>

                          <span className="text-[10px] text-neutral-500 mr-2">TRANCHE_0{t.tranche}_GATE</span>

                        </div>
                      </div>
                      <div className="flex items-center gap-4 text-[10px] pl-7 sm:pl-0">
                        <span className="text-amber-500 font-bold text-xs">{t.percentageUnlock}% UNLOCK</span>
                        <span className="px-2 py-0.5 font-bold tracking-widest bg-neutral-800 text-neutral-500">PENDING</span>
                      </div>
                    </div>
                  ))}
                </div>
              </div>
            )}

            {/* 3. Milestone Bonuses */}
            {vaultF.executiveLeverage && vaultF.executiveLeverage.gates && (
              <div className="mb-4">
                <div className="text-[10px] text-neutral-500 mb-4 tracking-widest border-b border-neutral-800 pb-2">
                  PREDICTIVE YIELD & MILESTONE BONUSES
                </div>
                <div className="space-y-2">
                  {vaultF.executiveLeverage.gates.map((g: any, idx: number) => {
                    const hasBonus = !!g.bonusEstimate;
                    if (!hasBonus) return null;
                    const formattedBonus = new Intl.NumberFormat("en-US", { style: "currency", currency: "USD", maximumFractionDigits: 0 }).format(g.bonusEstimate.amount);
                    return (
                      <div key={idx} className="flex flex-col sm:flex-row sm:items-center justify-between p-3 border border-neutral-800 bg-[#0d1117]">
                        <div className="flex items-center gap-3 mb-2 sm:mb-0">
                          <div className="w-4 h-4 border border-neutral-600 shrink-0"></div>
                          <div>
                            <span className="text-[10px] text-neutral-500 mr-2">BONUS_GATE</span>
                            <span className="font-bold text-neutral-300">{g.vector}</span>
                          </div>
                        </div>
                        <div className="flex items-center gap-4 text-[10px] pl-7 sm:pl-0">
                          <span className="text-blue-400 font-bold text-xs">{formattedBonus}</span>
                          <span className="px-2 py-0.5 font-bold tracking-widest bg-neutral-800 text-neutral-500">PENDING</span>
                        </div>
                      </div>
                    );
                  })}
                </div>
              </div>
            )}
          </div>
        </div>
      )}

      {vaultF && (
        <div className="mt-8">
          <InteractiveRoiDashboard />
        </div>
      )}

    </div>
  );
};
